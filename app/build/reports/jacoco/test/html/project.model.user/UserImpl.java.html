package project.model.user;

import project.model.payment.PaymentManager;
import project.model.product.ProductManager;

abstract class UserImpl implements User {
    protected String username;
    protected String password;
    protected PaymentManager paymentManager;
    protected UserManager userManager;
    protected ProductManager productManager;

    @Override
    public boolean injectPaymentManager(PaymentManager paymentManager) {
        if (paymentManager == null) {
            return false;
        }
        this.paymentManager = paymentManager;
        return true;
    }

    @Override
    public boolean injectUserManager(UserManager userManager) {
        if (userManager == null) {
            return false;
        }
        this.userManager = userManager;
        return true;
    }

    @Override
    public boolean injectProductManager(ProductManager productManager) {
        if (productManager == null) {
            return false;
        }
        this.productManager = productManager;
        return true;
    }

    @Override
    public String username() {
        return username;
    }

    @Override
    public boolean setUsername(String username) {
        this.username = username;
        return true;
    }

    @Override
    public String password() {
        return password;
    }

    @Override
    public boolean setPassword(String password) {
        this.password = password;
        return true;
    }

    @Override
    public boolean isAnonymous() {
        return username == null;
    }
}
