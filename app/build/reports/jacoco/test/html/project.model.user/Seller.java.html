package project.model.user;

import project.model.Manager;
import project.model.product.ProductContainer;

public class Seller extends UserImpl {

    @Override
    public Manager modify() {
        return productManager;
    }
    
    @Override
    public String list() {
        StringBuilder sb = new StringBuilder();
        for (ProductContainer pc : productManager.db()) {
            String s = String.format("Name: %s, Code: %s, Category: %s, Quantity: %d, Price: %d\n",
                pc.product().name(),
                pc.product().code(),
                pc.product().category(),
                pc.quantityLeft(),
                pc.product().price()
            );
            sb.append(s);
        }
        return sb.toString();
    }

    @Override
    public String summarise() {
        StringBuilder sb = new StringBuilder();
        for (ProductContainer pc : productManager.db()) {
            String s = String.format("%s; %s; %d\n",
                pc.product().code(),
                pc.product().name(),
                pc.quantitySold()
            );
            sb.append(s);
        }
        return sb.toString();
    }

}
