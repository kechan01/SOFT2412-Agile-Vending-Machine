package project.view;

import project.model.VendingMachine;

import java.io.Console;
import java.util.Scanner;

public class AppInterface {
    private final VendingMachine vendingMachine;
    private final SharedUI sharedUI;
    public AppInterface() {
        this.vendingMachine = new VendingMachine();
        this.sharedUI = new SharedUI();
    }

    public VendingMachine vendingMachine() {
        return vendingMachine;
    }

    public void run() {
        System.out.println("=============================================");
        System.out.println("        WELCOME TO VENDING MACHINE!");
        System.out.println("=============================================");

        // prints list of product options
        sharedUI.productOptions(vendingMachine.products());
        Scanner sc = new Scanner(System.in);

        while(true) {
           sharedUI.accountSelection();
           sharedUI.createAccount(sc, vendingMachine.getUsers());
            Console console = System.console();
            System.out.println(console);

        }

    }
}
