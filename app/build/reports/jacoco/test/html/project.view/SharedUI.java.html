    public void productTable(String category, List<ProductContainer> productContainerList) {
        System.out.println("---------------------------------------------");
        System.out.printf("%-45s %n", "CATEGORY : " + category);
        System.out.println("---------------------------------------------");
        System.out.printf("%-15s %-15s %-15s %n", "Name", "Price", "Quantity");
        System.out.println("---------------------------------------------");

        for (ProductContainer container : productContainerList) {
            if (container.product().category().equals(category)) {
                String name = container.product().name();
                double price = container.product().price();
                double quantity = container.quantityLeft();
                System.out.printf("%-15s $%-14s %-15s %n", name, price, quantity);
            }
        }
        System.out.println("---------------------------------------------");
    }
