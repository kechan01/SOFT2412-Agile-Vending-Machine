package project.model.user;

import project.model.Manager;
import project.model.payment.Transaction;

public class Owner extends UserImpl {

    @Override
    public Manager modify() {
        return userManager;
    }
    
    @Override
    public String list() {
        StringBuilder sb = new StringBuilder();
        for (User u : userManager.db().values()) {
            String s = u.getClass().getSimpleName();
            if (u.getClass().equals(Owner.class)) {
                s = "Admin";
            }
            sb.append(String.format("Username: %s, Role: %s\n", u.username(), s));
        }
        return sb.toString();
    }

    @Override
    public String summarise() {
        StringBuilder sb = new StringBuilder();
        for (Transaction t : paymentManager.transactions()) {
            if (t.cancellationReason() == null) {
                continue;
            }
            String role = null;
            if (t.user().isAnonymous()) {
                role = "Anonymous";
            } else {
                role = t.user().getClass().getSimpleName();
            }
            sb.append(String.format("Date/Time cancelled: %s, User: %s, Reasons: %s\n", t.date().toString(), role, t.cancellationReason()));
        }
        return sb.toString();
    }

    
}
