package project.model.user;

import java.util.List;
import java.util.Map;

import project.model.Manager;
import project.model.payment.Transaction;

public class Cashier extends UserImpl {

    @Override
    public Manager modify() {
        return paymentManager;
    }

    @Override
    public String list() {
        StringBuilder sb = new StringBuilder();
        Map<Double, Integer> map = paymentManager.cashManager().register();
        for (Double d : map.keySet()) {
            sb.append(String.format("Denomination: %d, Quantity: %d\n", d, map.get(d)));
        }
        return sb.toString();
    }

    @Override
    public String summarise() {
        StringBuilder sb = new StringBuilder();
        List<Transaction> ls = paymentManager.transactions();
        for (Transaction t : ls) {
            sb.append(String.format("Date/Time: %s, Item Sold: %s, Amount Paid: %d, Returned Change: %d, Payment Method: %s\n",
                t.date().toString(),
                t.product().name(),
                t.amount(),
                t.change(),
                t.paymentMethod().toString()
            ));
        }
        return sb.toString();
    }

    
}
